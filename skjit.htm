const int redLed = 2;

void setup() {
  pinMode(redLed, OUTPUT);
}

void loop() {
  digitalBlink(500, 10);
}

//Nytt inngangsargument, repeats.
void digitalBlink(int waitTime, int repeats) {
  int counter = 0;

  //Løkke for å gjenta repeats antall ganger. 
  //Deklarerer lokal tellevariabel.
  while(counter < repeats) {
    digitalWrite(redLed, HIGH);
    delay(waitTime);
    digitalWrite(redLed, LOW);
    delay(waitTime);

    counter += 1;
  }
}
